### 텐서플로우 설치 명령
python -m pip install --upgrade pip  <br>
pip3 install tensorflow==2.10.0 <br>


### 텐서플로우 가상 환경 설치 명령
conda create -n tf210 python=3.10 anaconda <br>
conda activate tf210 <br>
conda update pip <br>
pip3 install tensorflow==2.10.0 <br>
